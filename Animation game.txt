
#include <raylib.h>

int main ()
{
    const int screen_width = 800;
    const int screen_height = 600;

    int velocity = 0;
    int gravity = 1;
    int jump = 10;
    bool isair = false;

    InitWindow(screen_width,screen_height, "Texture 2D Game");

    Texture2D scarfy;
    scarfy = LoadTexture("Textures/scarfy");
    Rectangle scarfyrec;
    scarfyrec.width = scarfy.width/6;
    scarfyrec.height = scarfy.height;
    scarfyrec.x = 0;
    scarfyrec.y = 0;
    Vector2 scarfypos;
    scarfypos.x = screen_width/2;
    scarfypos.y = screen_height - scarfy.height;
    SetTargetFPS(60);
    while(!WindowShouldClose())
    {
        BeginDrawing();
        ClearBackground(WHITE);

        if(scarfypos.y >= screen_height- scarfy.height  )
        {
            velocity = 0;
            isair = false;
        }
        else
        {
            velocity += gravity;
            isair = true;
        }

        if(IsKeyPressed(KEY_SPACE) && !isair)
        {
            velocity -= jump;
        }

        scarfypos.y += velocity;

      //  DrawTextureRec(scarfy.scarfyrec,scarfypos,WHITE);
        EndDrawing();
    }
UnloadTexture(scarfy);
CloseWindow();
}
